# Sui Wallet Kit Demo

import {
    WalletKitProvider,
    ConnectButton,
    useWalletKit,
} from '@mysten/wallet-kit';

export const Demo = () => {
    const { currentAccount, currentWallet } = useWalletKit();
    if (!currentAccount) return null;
    return (
        <div>
            <div>
                <strong>Current Account:</strong> {currentAccount?.address}
            </div>
            <div>
                <strong>Current Wallet:</strong> {currentWallet?.name}
            </div>
        </div>
    );
};

<WalletKitProvider>
    <div style={{ display: 'flex', justifyContent: 'center', padding: 16 }}>
        <ConnectButton />
    </div>
    <Demo />
</WalletKitProvider>
